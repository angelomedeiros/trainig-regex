      package cod3r;

      /*
       * Imprimir nota do aluno:
       * - Maior ou igual a sete, aprovado;
       * - Menor que sete, reprovado.
       *
       */

       public class Nota {
         public static void main(String[] args) {
           int nota = 7;

           if (nota >= 7) {
             System.out.println("Aprovado");
           } else {
             System.out.println("Reprovado");
           }
         }
       }
